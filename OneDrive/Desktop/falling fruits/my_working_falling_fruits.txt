    <script>
        let appleScore = 0;
        let orangeScore = 0;
        let bananaScore = 0;

        document.querySelectorAll('.item').forEach(item => {
            item.addEventListener('click', function(e) {
                const fruit = e.target.classList[0];
                if (fruit === 'apple') {
                    appleScore++;
                    document.getElementById('appleScore').textContent = appleScore;
                } else if (fruit === 'orange') {
                    orangeScore++;
                    document.getElementById('orangeScore').textContent = orangeScore;
                } else if (fruit === 'banana') {
                    bananaScore++;
                    document.getElementById('bananaScore').textContent = bananaScore;
                }
            });
        });
    </script>
